Base

config --global user.name blabla - ім'я користувача.
git config --global user.email blabal@bla - електронна пошта користувача.
git init - ініціює порожній репозиторій в поточній дерикторії.
git status - стан файлів в робочій папці.
git add - вибирає файли для коміта.
git commit - зберігає всі файли що ми вибрали в коміт.
git log - перегляд історії комітів.
git checkout - перехід на іншу гілку.
git branch - менеджер гілок: перечислює, створює, видаляє, перейменовує гілки і тд.
git push - встановлює зв'язок з віддаленим репозиторієм.
git pull - забирає зміни з віддаленого репозиторію на локальний.
git merge - зливає гілки в одну.
git clone - копіює репозиторій.

.git ignor - файл, який містить файли чи папки які потрібно приховати від Git.

Advanced

git revert - скасовує коміти створені попередньо в файлі.
git reset - скасовує існуючі коміти.
git rebase - об'єднує зміни зроблені в одній гілці з іншою гілкою.
git cherry-pick - використовує коміти іншої гілки без повного злиття гілок.
git stash - ховає зміни в файлах, зберігає їх окремо, пізніше їх можна повернути.
